function identifikasiSegitiga(a, b, c) {
    // Validasi input
    if (a <= 0 || b <= 0 || c <= 0) {
        return "Error: Nilai sisi harus positif";
    }
    
    // Periksa ketentuan segitiga
    if (a + b <= c || a + c <= b || b + c <= a) {
        return "Error: Bukan segitiga valid";
    }
    
    // Klasifikasi jenis segitiga
    if (a === b && b === c) {
        return "Segitiga Sama Sisi";
    } else if (a === b || b === c || a === c) {
        return "Segitiga Sama Kaki";
    } else {
        return "Segitiga Sembarang";
    }
}
